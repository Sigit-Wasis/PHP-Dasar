CREATE TABLE Karyawan (
id INT AUTO_INCREMENT PRIMARY KEY,
nama_depan VARCHAR(50) NOT NULL,
jabatan VARCHAR(50),
gaji INT(255)
);

-- untuk Menampilkan data Karyawan --
SELECT * FROM Karyawan;

-- untuk menampilkan nama dan gaji karyawan yang berjabatan Programmer --
SELECT nama_depan, gaji FROM Karyawan WHERE jabatan = 'Programer';

-- untuk Menampilkan nama karyawan yang memiliki gaji lebih dari Rp 8.000.000.--
SELECT nama_depan FROM Karyawan WHERE gaji > 8000000;

-- untuk menghitung total jumlah gaji karyawan --
SELECT SUM(gaji) AS total_gaji FROM Karyawan;

-- untuk menemukan karyawan dengan gaji tertinggi --
SELECT nama_depan, jabatan, gaji FROM Karyawan WHERE gaji = (SELECT MAX(gaji) FROM Karyawan);

-- untuk mengurutkan data karyawan berdasarkan nama (ascending) --
SELECT * FROM Karyawan ORDER BY nama_depan ASC;
